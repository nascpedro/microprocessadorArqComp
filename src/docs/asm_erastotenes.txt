--preenchemos os enderecos da RAM[2] a RAM[31] com 1 a 31, respectivamente.
        LD R1, 2       -- R1 é o contador/ponteiro inicial
LOOP_INICIALIZACAO:
        MOV A, R1      
        SW A, (R1)     -- Grava o valor na RAM 
        
        ADDI A, 1      -- Incrementa ponteiro
        MOV R1, A      
        
        -- fim do loop com SLR
        SLR A, R1      -- Se R1 < 32, A = 0. Se R1 >= 32, A > 0.
        CMPI A, 0      -- Compara o resultado com 0
        BHI FIM_INICIALIZACAO   -- Se A > 0 (chegou no 32), sai do loop pulando para FIM_INICIALIZACAO
        JMP LOOP_INICIALIZACAO  -- Se A == 0, repete o loop
FIM_INICIALIZACAO:

-- crivo do 2
        LD R1, 4       -- Começa no primeiro múltiplo (4)
        LD R0, 0       -- R0 fixo em zero para zerar valores
LOOP_DOIS:
        MOV A, R0      
        SW A, (R1)     -- RAM[R1] = 0 (Apaga o não-primo)
        
        MOV A, R1
        ADDI A, 2      -- Avança de 2 em 2
        MOV R1, A
        
        SLR A, R1
        CMPI A, 0
        BHI FIM_DOIS
        JMP LOOP_DOIS
FIM_DOIS:

-- crivo do 3
        LD R1, 6       -- Começa no primeiro múltiplo (6)
LOOP_TRES:
        MOV A, R0      -- R0 ainda guarda 0
        SW A, (R1)     
        
        MOV A, R1
        ADDI A, 3      -- Avança de 3 em 3
        MOV R1, A
        
        SLR A, R1
        CMPI A, 0
        BHI FIM_TRES
        JMP LOOP_TRES
FIM_TRES:

-- crivo do 5
        LD R1, 10      -- Começa no primeiro múltiplo (10)
LOOP_CINCO:
        MOV A, R0      
        SW A, (R1)     
        
        MOV A, R1
        ADDI A, 5      -- Avança de 5 em 5
        MOV R1, A
        
        SLR A, R1
        CMPI A, 0
        BHI FIM_CINCO
        JMP LOOP_CINCO
FIM_CINCO:

-- exibe os primos encontrados
        LD R1, 2       -- Volta ponteiro para o início
LOOP_EXIBE:
        LW R2, (R1)    -- Lê a RAM no endereço atual
        MOV A, R2      -- Move para o Acumulador para vermos pela ULA
        
        MOV A, R1
        ADDI A, 1      -- Próximo endereço
        MOV R1, A
        
        SLR A, R1
        CMPI A, 0
        BHI FIM      -- Se bateu no 32, termina o programa
        JMP LOOP_EXIBE
FIM:
        JMP 0          -- Trava o PC em loop infinito
